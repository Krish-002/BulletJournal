package cs3500.pa05.model;

public enum Day {
  MONDAY, TUESDAY, WEDNESDAY, THURSDAY, FRIDAY, SATURDAY, SUNDAY;

    public static Day fromString(String s) {
        switch (s) {
        case "MONDAY":
            return MONDAY;
        case "TUESDAY":
            return TUESDAY;
        case "WEDNESDAY":
            return WEDNESDAY;
        case "THURSDAY":
            return THURSDAY;
        case "FRIDAY":
            return FRIDAY;
        case "SATURDAY":
            return SATURDAY;
        case "SUNDAY":
            return SUNDAY;
        default:
            throw new IllegalArgumentException("Invalid day");
        }
    }
}
