package cs3500.pa05.model;

import cs3500.pa05.model.records.EventJson;

public class Event {
  String name;
  String description;
  Day day;
  Time start;
  int duration;

  public Event(String name, String description, Day day, Time start, int duration) {
    this.name = name;
    this.description = description;
    this.day = day;
    this.start = start;
    this.duration = duration;
  }

  public EventJson toEventJson() {
    return new EventJson(this.name, this.description, this.day.toString(), this.start.toTimeJson(), this.duration);
  }

  public Day getDay() {
    return day;
  }

  @Override
    public String toString() {
        return this.name + ": " + this.description + " " + "Start Time: " +start.toString() + " Duration: " + duration;
    }
}
